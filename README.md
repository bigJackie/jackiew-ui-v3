## install

```javascript
npm install jackiew-ui-v3
```

## Quick Start

```javascript
import JUI from "jackiew-ui";
import "jackiew-ui/jui.css";
Vue.use(JUI);

// or
import {
  JAppBar,
  JButton,
  // ...
} from "jackiew-ui";
Vue.component(JAppBar.name, JAppBar);
Vue.component(JButton.name, JButton);
```
